# winget 支持安装软件列表

[[TOC]]

## 开发软件

```bash
# Docker Desktop
winget install -h Docker.DockerDesktop
# Windows Terminal
winget install -h Microsoft.WindowsTerminal
# Google Chrome
winget install -h Google.Chrome
# Python 3.10 / 3.11
winget install -h Python.Python.3.10
# Visual Studio Code
winget install -h Microsoft.VisualStudioCode
# Visual Studio 2022
winget install -h Microsoft.VisualStudio.2022.Community
# 7z
winget install -h 7zip.7zip
# Git
winget install -h Git.Git
# Chocolatey
winget install -h Chocolatey.Chocolatey
# Oracle JDK / OpenJDK
winget install -h Oracle.JDK.17
# GoLang
winget install -h GoLang.Go
# Rust
winget install -h Rustlang.Rust.MSVC
# PowerToys
winget install -h Microsoft.PowerToys
# Everything
winget install -h voidtools.Everything
# CMake
winget install -h Kitware.CMake
# Apifox
winget install -h Apifox.Apifox
# DevToys
winget install -h 9PGCV4V3BK4W
# 微软电脑管家
winget install -h 9PM860492SZD
# Office 365
winget install -h 9WZDNCRD29V9
# Navicat
winget install -h PremiumSoft.NavicatPremium
# JetBrains IDEA
winget install -h JetBrains.IntelliJIDEA.Ultimate
# JetBrains WebStorm
winget install -h JetBrains.WebStorm
# TreeSize Free
winget install -h JAMSoftware.TreeSize.Free
```
